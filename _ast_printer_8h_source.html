197 template<class N>
198 void printNamedNodeList(
199 const typename NamedNode<N, Ident::OptPR>::List & els,
200 const std::string & opening,
201 const std::string & closing,
202 const std::string & sep
203 ) {
204 log.raw(opening);
205
206 for (size_t i = 0; i < els.size(); i++) {
207 const auto & el = els.at(i);
208 el.name.then([&](const Ident::PR & name) {
209 name.autoAccept(*this);
210 log.raw(": ");
211 });
212
213 el.node.autoAccept(*this);
214
215 if (i < els.size() - 1) {
216 log.raw(sep);
217 }
218 }
219
220 log.raw(closing);
221 }

225 template<class T>
226 void basePrintDelim(
227 const std::vector<T> & elements,
228 const std::string & begin,
229 const std::string & end,
230 const std::string & delim,
231 uint8_t chopTH,
232 const std::function<void(const T &)> & cb
233 ) {
234 const auto chop = elements.size() > chopTH;
235 if (not begin.empty()) {
236 log.raw(begin);
237 }
238 if (chop) {
239 log.nl();
240 incIndent();
241 }
242 for (size_t i = 0; i < elements.size(); i++) {
243 if (chop) {
244 printIndent();
245 }
246 cb(elements.at(i));
247 if (i < elements.size() - 1) {
248 log.raw(delim);
249 }
250 }
251 if (chop) {
252 log.nl();
253 decIndent();
254 printIndent();
255 }
256 if (not end.empty()) {
257 log.raw(end);
258 }
259 }

265 const std::string & end = "",
266 const std::string & delim = ", ",
267 uint8_t chopTH = DEFAULT_CHOP_THRESHOLD
268 ) {
