130
131 template<class T>
132 void basePrintDelim(
133 const std::vector<T> & elements,
134 const std::string & begin,
135 const std::string & end,
136 const std::string & delim,
137 uint8_t chopTH,
138 const std::function<void(const T&)> & cb
139 ) {
140 const auto chop = elements.size() > chopTH;
141 if (not begin.empty()) {
142 log.raw(begin);
143 }
144 if (chop) {
145 log.nl();
146 incIndent();
147 }
148 for (size_t i = 0; i < elements.size(); i++) {
149 if (chop) {
150 printIndent();
151 }
152 cb(elements.at(i));
153 if (i < elements.size() - 1) {
154 log.raw(delim);
155 }
156 }
157 if (chop) {
158 log.nl();
159 decIndent();
160 printIndent();
161 }
162 if (not end.empty()) {
163 log.raw(end);
164 }
165 }

167 template<class T>
168 void printDelim(
169 const std::vector<T> & elements,
170 const std::string & begin = "",
171 const std::string & end = "",
172 const std::string & delim = ", ",
173 uint8_t chopTH = DEFAULT_CHOP_THRESHOLD
174 ) {
175 basePrintDelim<T>(elements, begin, end, delim, chopTH, [&](const T & el) -> void {
176 if constexpr (dt::is_ptr_like<T>::value) {
177 el->accept(*this);
178 } else {
179 el.accept(*this);
180 }
181 });
182 }
